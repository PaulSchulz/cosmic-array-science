  ssh pi@$CAHOST mv git/cosmic-array/cosmic-array.log git/cosmic-array.log.$DATESTAMP
  ssh pi@$CAHOST reboot

Get files:
  scp pi@$CAHOST:git/cosmic-array/cosmic-array.log.* .

  date -d @1515142282

Combine the files
  cat cosmic-array.log.[0-9]* > cosmic-array.log.combined
